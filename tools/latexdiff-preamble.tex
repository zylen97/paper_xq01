%DIF CUSTOM PREAMBLE - Blue Text for Additions, Red Strikethrough for Deletions
%DIF Designed for elsarticle + pdfLaTeX workflow
%DIF Citations are extracted from \DIFdel{} in post-processing (make-diff.sh)
%DIF so they can break across lines while the surrounding text keeps strikethrough.

\RequirePackage[normalem]{ulem}
\RequirePackage{xcolor}

% Color definitions
\definecolor{addcolor}{RGB}{0,0,200}      % Blue for additions
\definecolor{delcolor}{RGB}{200,50,50}    % Red for deletions

% Additions -> blue text
\providecommand{\DIFadd}[1]{{\color{addcolor}#1}}
% Deletions -> red strikethrough
\providecommand{\DIFdel}[1]{{\color{delcolor}\sout{#1}}}

% Paragraph-level markers (begin/end pairs)
\providecommand{\DIFaddbegin}{}
\providecommand{\DIFaddend}{}
\providecommand{\DIFdelbegin}{}
\providecommand{\DIFdelend}{}
\providecommand{\DIFmodbegin}{}
\providecommand{\DIFmodend}{}

% Float-safe versions (inside tables/figures)
\providecommand{\DIFaddFL}[1]{\DIFadd{#1}}
\providecommand{\DIFdelFL}[1]{\DIFdel{#1}}
\providecommand{\DIFaddbeginFL}{}
\providecommand{\DIFaddendFL}{}
\providecommand{\DIFdelbeginFL}{}
\providecommand{\DIFdelendFL}{}
